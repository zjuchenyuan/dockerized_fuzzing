1/**
2 * Handling of Replay Gain information in ID3v2 tags.
3 *
4 * We store Replay Gain data in RVA2 frames inside the ID3v2.4 tag.
5 * Album and track gain are written as two separate frames.
6 * We use the RVA2 frame as follows:
7 *
8 * Identification string: either "track" or "album";
9 * Channel number: always 1 (master volume);
10 * Volume adjustment: recommended gain relative to 89 dB standard;
11 * Bits representing peak: always 16;
12 * Peak volume: max absolute sample value relative to full scale
13 *
14 * The meaning of the RVA2 peak volume field is not specified in ID3v2.4.
15 * We follow the interpretation of QuodLibet/mutagen: Peak volume is the
16 * maximum absolute sample value relative to full scale, before application
17 * of the volume adjustment, stored as an unsigned fixed point value with
18 * 1 integer bit and (peakbits-1) fractional bits.
19 *
20 * In addition to standard Replay Gain data, we also store mp3gain-specific
21 * fields in TXXX frames. The description string of such frames starts with
22 * "MP3GAIN_". These frames are only needed when mp3gain updates the encoded
23 * audio volume in MP3 stream (-r and -a command line options).
24 *
25 * We read tag data in ID3v2.2, ID3v2.3 or ID3v2.4 format, from either
26 * the beginning or the end of the file. Extended tag headers are ignored
27 * and removed; compressed frames are ignored but preserved. No workarounds
28 * are attempted for invalid ID3v2.4 tags.
29 *
30 * When writing/updating tag data, we always write a single ID3v2.4 tag
31 * at the beginning of the file, fully unsynchronized. If the original
32 * file had an ID3v2.2 or ID3v2.3 tag, it is upgraded to ID3v2.4. If the
33 * original file had only an appended tag, it is moved to the beginning of
34 * the file. If the original file contained multiple tags, we update only
35 * the first tag and ignore the rest (this is bad, but it's probably a
36 * rare case). If the original file had only an ID3v1 tag, it is copied
37 * to a new ID3v2 tag and the ID3v1 tag is left as it was.
38 *
39 * See: http://www.id3.org/id3v2.4.0-structure
40 * http://www.id3.org/id3v2.4.0-frames
41 */

315/**
316 * Decode an RVA2 frame (only track/album gain, only master channel).
317 *
318 * Store gain information in the info structure, unless info == NULL.
319 * Return 1 if the frame is an RVA2 frame with track/album gain, 0 otherwise.
320 */
321static int id3_decode_rva2_frame(const struct ID3v2FrameStruct *frame, struct MP3GainTagInfo *info)
322{
323 unsigned long p;
324 int is_album_gain, channel, peakbits;
325 double gain, peak;
326
327 /* Ignore non-RVA2 frames. */
328 if (memcmp(frame->frameid, "RVA2", 4) != 0)
329 return 0;
330
331 p = frame->hskip;
332
333 /* Check identification string; we understand only "track" and "album". */
334 if (p + 6 <= frame->len &&
335 (memcmp(frame->data + p, "track\0", 6) == 0 ||
336 memcmp(frame->data + p, "TRACK\0", 6) == 0)) {
337 is_album_gain = 0;
338 p += 6;
339 } else if (p + 6 <= frame->len &&
340 (memcmp(frame->data + p, "album\0", 6) == 0 ||
341 memcmp(frame->data + p, "ALBUM\0", 6) == 0)) {
342 is_album_gain = 1;
343 p += 6;
344 } else {
345 return 0;
346 }
347
348 /* Process per-channel data. */
349 while (p + 4 <= frame->len) {
350 /*
351 * p+0 : channel number
352 * p+1, p+2 : 16-bit signed BE int = adjustment * 512
353 * p+3 : nr of bits representing peak
354 * p+4 ... : unsigned multibyte BE int = peak sample * 2**(peakbits-1)
355 */
356 channel = frame->data[p];
357 gain = (double)(((signed char)(frame->data[p+1]) << 8) | frame->data[p+2]) / 512.0;
358 peakbits = frame->data[p+3];
359 if (p + 4 + (peakbits + 7) / 8 > frame->len)
360 break;
361 peak = 0;
362 if (peakbits > 0) peak += (double)(frame->data[p+4]);
363 if (peakbits > 8) peak += (double)(frame->data[p+5]) / 256.0;
364 if (peakbits > 16) peak += (double)(frame->data[p+6]) / 65536.0;
365 if (peakbits > 0) peak = peak / (double)(1 << ((peakbits - 1) & 7));
366 p += 4 + (peakbits + 7) / 8;
367 if (channel == 1) { /* channel == master volume channel */
368 if (info) {
369 if (is_album_gain) {
370 info->haveAlbumGain = 1;
371 info->albumGain = gain;
372 info->haveAlbumPeak = (peakbits > 0);
373 info->albumPeak = peak;
374 } else {
375 info->haveTrackGain = 1;
376 info->trackGain = gain;
377 info->haveTrackPeak = (peakbits > 0);
378 info->trackPeak = peak;
379 }
380 }
381 }
382 }
383
384 return 1;
385}

488/**
489 * Read an ID3v2 tag from the current position in the MP3 file.
490 *
491 * Return 1 on success, 0 if no tag is found, or a negative error code
492 * if the tag can not be processed.
493 */
494static int id3_parse_v2_tag(FILE *f, struct ID3v2TagStruct *tag)
495{
496 unsigned char buf[12], frameid[4];
497 unsigned int fflags;
498 unsigned long dlen, flen, fhskip, p, k;
499 unsigned char *tagdata;

504 if (fread(buf, 1, 10, f) != 10)
505 return 0;
506
507 /* Check 'ID3' signature. */
508 if (memcmp(buf, "ID3", 3) != 0)
509 return 0;

511 DBG(("DEBUG: Found ID3v2 tag at offset %lu\n", tag->offset));
512
513 /* Check version and flags. */
514 switch (buf[3]) {
515 case 2: /* ID3v2.2 */
516 if (buf[5] & (~(TAGFL_UNSYNC(0x80))))
517 return M3G_ERR_TAGFORMAT-8; /* unknown flags */
518 break;
519 case 3: /* ID3v2.3 */
520 if (buf[5] & (~(TAGFL_UNSYNC(0x80) | TAGFL_EXTHDR(0x40) | TAGFL_EXPR(0x20))))
521 return M3G_ERR_TAGFORMAT-8; /* unknown flags */
522 break;
523 case 4: /* ID3v2.4 */
524 if (buf[5] & (~(TAGFL_UNSYNC(0x80) | TAGFL_EXTHDR(0x40) | TAGFL_EXPR(0x20) | TAGFL_FOOTER(0x10))))
525 return M3G_ERR_TAGFORMAT-8; /* unknown flags */
526 break;
527 default:
528 return M3G_ERR_TAGFORMAT-8; /* unknown version */
529 }
530
531 /* Check length. */
532 dlen = id3_get_syncsafe_int(buf + 6);
533 if (dlen == SYNCSAFE_INT_BAD(0xffffffff))
534 return M3G_ERR_TAGFORMAT-8;
535
536 /* Fill tag structure. */
537 tag->flags = buf[5];
538 tag->version = (buf[3] << 8) | buf[4];
539 tag->length = 10 + dlen + ((tag->flags & TAGFL_FOOTER(0x10)) ? 10 : 0);
540 tag->frames = NULL((void*)0);

542 DBG((" version=%04x length=%lu flags=%02x\n", tag->version, tag->length, tag->flags));
543

546 if (fread(tagdata, 1, dlen, f) != dlen)

550 if ((tag->version >> 8) != 4 && (tag->flags & TAGFL_UNSYNC(0x80)) != 0) {

552 }
553
554 /* Skip extended header. */
555 p = 0;
556 if ((tag->flags & TAGFL_EXTHDR(0x40)) != 0) {
557 DBG((" skip extended header\n"));
558 if (p + 6 > dlen)
559 goto badtag;
560 if ((tag->version >> 8) == 4) {
561 /* Skip ID3v2.4 extended header */
562 k = id3_get_syncsafe_int(tagdata + p);
563 if (k == SYNCSAFE_INT_BAD(0xffffffff) || k > dlen)
564 goto badtag;
565 p += k;
566 } else if ((tag->version >> 8) == 3) {
567 /* Skip ID3v2.3 extended header */
568 k = id3_get_int32(tagdata + p);
569 if (k > dlen)
570 goto badtag;
571 p += 4 + k;
572 }
573 }
574
575 /* Scan frames. */
576 pframe = &(tag->frames);
577 while (p < dlen && tagdata[p] != '\0') {
578
579 /* Decode frame header. */
580 switch (tag->version >> 8) {
581
582 case 2: /* ID3v2.2 */
583 if (p + 5 > dlen)
584 goto badtag;
585 memset(frameid, 0, 4);
586 for (k = 0; upgrade_id3v22_table[k].id_new[0]; k++) {
587 if (memcmp(tagdata + p, upgrade_id3v22_table[k].id_v22, 3) == 0) {
588 memcpy(frameid, upgrade_id3v22_table[k].id_new, 4);
589 break;
590 }
591 }
592 flen = (tagdata[p+3] << 16) | (tagdata[p+4] << 8) | tagdata[p+5];
593 fflags = 0;
594 if (flen > dlen)
595 goto badtag;
596 p += 6;
597 break;
598
599 case 3: /* ID3v2.3 */
600 if (p + 10 > dlen)
601 goto badtag;
602 memcpy(frameid, tagdata + p, 4);
603 flen = id3_get_int32(tagdata + p + 4);
604 fflags = (tagdata[p+8] << 7) & 0xff00;
605 if (tagdata[p+9] & 0x80) fflags |= FRAMEFL_COMPR(0x0008) | FRAMEFL_DLEN(0x0001);
606 if (tagdata[p+9] & 0x40) fflags |= FRAMEFL_CRYPT(0x0004);
607 if (tagdata[p+9] & 0x20) fflags |= FRAMEFL_GROUP(0x0040);
608 if (tagdata[p+9] & 0x1f) fflags |= FRAMEFL_BAD(0x00f0);
609 if (flen > dlen)
610 goto badtag;
611 p += 10;
612 break;
613
614 case 4: /* ID3v2.4 */
615 if (p + 10 > dlen)
616 goto badtag;
617 memcpy(frameid, tagdata + p, 4);
618 flen = id3_get_syncsafe_int(tagdata + p + 4);
619 fflags = (tagdata[p+8] << 8) | tagdata[p+9];
620 if (flen == SYNCSAFE_INT_BAD(0xffffffff) || flen > dlen)
621 goto badtag;
622 p += 10;
623 break;
624
625 default: /* unreachable */
626 goto badtag;
627 }
628
629 if (p + flen > dlen)
630 goto badtag;
631
632 DBG((" got frameid=%.4s fflags=%04x\n", frameid, fflags));
633
634 /* Drop unsupported frame types. */
635 if (frameid[0] == '\0' ||
636 memcmp(frameid, "RVAD", 4) == 0 ||
637 memcmp(frameid, "RGAD", 4) == 0 ||
638 memcmp(frameid, "XRVA", 4) == 0) {
639 DBG((" drop unsupported frame\n"));
640 p += flen;
641 continue;
642 }
643
644 /* Drop v2.3 frames which we can not upgrade. */
645 if ((tag->version >> 8) == 3 && (fflags & (FRAMEFL_CRYPT(0x0004) | FRAMEFL_BAD(0x00f0))) != 0) {
646 DBG((" drop non-upgradable frame\n"));
647 p += flen;
648 continue;
649 }
650
651 /* Drop frames that are too short for their flags. */
652 fhskip = (fflags & FRAMEFL_GROUP(0x0040) ? 1 : 0) +
653 (fflags & FRAMEFL_CRYPT(0x0004) ? 1 : 0) +
654 (fflags & FRAMEFL_DLEN(0x0001) ? 4 : 0);
655 if (fhskip > flen) {
656 DBG((" drop short frame\n"));
657 p += flen;
658 continue;
659 }
660
661 /* Drop v2.2 PIC frames that are too short. */
662 if ((tag->version >> 8) == 2 && memcmp(frameid, "APIC", 4) == 0 && flen < 6) {
663 DBG((" drop short PIC frame\n"));
664 p += flen;
665 continue;
666 }
667
668 /* Rename TYER (Year) to TDRC (Recording time) */
669 if (memcmp(frameid, "TYER", 4) == 0)
670 memcpy(frameid, "TDRC", 4);
671
672 /* Drop frames that want to be dropped on tag alteration. */
673 if (fflags & FRAMEFL_TAGALTER(0x4000)) {
674 DBG((" drop FRAMEFL_TAGALTER frame\n"));
675 p += flen;
676 continue;
677 }
678
679 /* Allocate frame structure. */
680 frame = malloc(sizeof(struct ID3v2FrameStruct));
681 frame->next = NULL((void*)0);
682 memcpy(frame->frameid, frameid, 4);
683 frame->flags = fflags;
684 frame->hskip = fhskip;
685 *pframe = frame;
686 pframe = &(frame->next);
687
688 /* Copy frame data. */
689 if ((tag->version >> 8) == 4 && (fflags & FRAMEFL_UNSYNC(0x0002)) != 0) {
690 /* This frame is unsynchronized; decode it now. */
691 frame->data = malloc(flen);
692 k = id3_get_unsync_data(frame->data, tagdata + p, flen);
693 frame->len = k;
694 p += flen;
695 } else if ((tag->version >> 8) == 2 && memcmp(frameid, "APIC", 4) == 0) {
696 /* APIC frame format differs from PIC frame format */
697 frame->data = malloc(flen + 12);
698 frame->data[0] = tagdata[p];
699 k = 1;
700 if (memcmp(tagdata + p + 1, "PNG", 3) == 0) {
701 memcpy(frame->data + k, "image/png", strlen("image/png") + 1);
702 k += strlen("image/png") + 1;
703 } else if (memcmp(tagdata + p + 1, "JPG", 3) == 0) {
704 memcpy(frame->data + k, "image/jpeg", strlen("image/jpeg") + 1);
705 k += strlen("image/jpeg") + 1;
706 } else if (tagdata[p+1] == '\0') {
707 memcpy(frame->data + k, tagdata + p + 1, 3);
708 frame->data[k+3] = '\0';
709 k += 4;
710 }
711 memcpy(frame->data + k, tagdata + p + 4, flen - 4);
712 frame->len = k + flen - 4;
713 p += flen;
714 } else {
715 /* Normal case, just copy the data. */
716 frame->data = malloc(flen);
717 memcpy(frame->data, tagdata + p, flen);
718 frame->len = flen;
719 p += flen;
720 }
721
722 /* Reorder flag parameters of upgraded v2.3 frames. */
723 if ((tag->version >> 8) == 3 && (fflags & FRAMEFL_DLEN(0x0001)) != 0) {
724 k = id3_get_int32(frame->data);
725 if (fflags & FRAMEFL_GROUP(0x0040)) {
726 frame->data[0] = frame->data[4];
727 id3_put_syncsafe_int(frame->data + 1, k);
728 } else {
729 id3_put_syncsafe_int(frame->data, k);
730 }
731 }
732
733 }
734
735 if (p > dlen)
736 goto badtag;
737
738 free(tagdata);
739 return 1;
740
741badtag:
742 free(tagdata);
743 id3_release_frames(tag->frames);
744 return M3G_ERR_TAGFORMAT-8;
745}

748/**
749 * Write an ID3v2 tag at the current position in the MP3 file.
750 *
751 * Tags are always written in ID3v2.4 format.
752 * Tags are always written completely unsynchronized,
753 * without extended header, padded up to an integer multiple of 2 KB.
754 * We don't write a tag footer (not supported by QuodLibet).
755 *
756 * Return 1 on success, 0 if the tag contains no frames,
757 * or a negative error code.
758 */
759static int id3_write_tag(FILE *f, struct ID3v2TagStruct *tag)
760{
761 unsigned long dlen, p, k;
762 unsigned char *tagdata;
763 struct ID3v2FrameStruct *frame;
764
765 DBG(("DEBUG: Writing ID3v2 tag\n"));
766
767 /* Do not write a tag with zero frames. */
768 if (tag->frames == NULL((void*)0))
769 return 0;
770
771 /* Compute raw total length of tag. */
772 dlen = 10; /* tag header */
773 for (frame = tag->frames; frame; frame = frame->next) {
774 dlen += 10;
775 dlen += id3_put_unsync_data(NULL((void*)0), frame->data, frame->len);
776 }
777 dlen = (dlen + 2047) & (~2047); /* padding */
778
779 DBG((" length=%lu\n", dlen));
780
781 /* Allocate buffer and fill with zeros. */
782 tagdata = calloc(dlen, sizeof(unsigned char));
783
784 /* Prepare tag header. */
785 tagdata[0] = 'I';
786 tagdata[1] = 'D';
787 tagdata[2] = '3';
788 tagdata[3] = 4;
789 tagdata[4] = 0;
790 tagdata[5] = TAGFL_UNSYNC(0x80) | (tag->flags & TAGFL_EXPR(0x20));
791 id3_put_syncsafe_int(tagdata + 6, dlen - 10);
792 p = 10;
793
794 /* Prepare frames. */
795 for (frame = tag->frames; frame; frame = frame->next) {
796 unsigned long fflags = frame->flags & (~FRAMEFL_UNSYNC(0x0002));
797 memcpy(tagdata + p, frame->frameid, 4);
798 k = id3_put_unsync_data(tagdata + p + 10, frame->data, frame->len);
799 id3_put_syncsafe_int(tagdata + p + 4, k);
800 if (k != frame->len) fflags |= FRAMEFL_UNSYNC(0x0002);
801 tagdata[p+8] = (fflags >> 8) & 0xff;
802 tagdata[p+9] = fflags & 0xff;
803 p += 10 + k;
804 DBG((" write frameid=%.4s length=%lu\n", frame->frameid, 10 + k));
805 }
806
807 /* Write the whole thing. */
808 if (fwrite(tagdata, 1, dlen, f) != dlen) {
809 free(tagdata);
810 return M3G_ERR_WRITE-7;
811 }
812
813 free(tagdata);
814 return 1;
815}

818/**
819 * Read an ID3v1 tag from the current position in the MP3 file.
820 *
821 * Return 1 on success, 0 if no tag is found.
822 */
823static int id3_parse_v1_tag(FILE *f, struct ID3v2TagStruct *tag)
824{
825 unsigned char buf[128];
826 char sbuf[32];
827
828 struct ID3v2FrameStruct **pframe;
829
830 /* Read tag */
831 tag->offset = ftell(f);
832 if (fread(buf, 1, 128, f) != 128)
833 return 0;
834
835 /* Check 'TAG' signature. */
836 if (memcmp(buf, "TAG", 3) != 0)
837 return 0;
838
839 DBG(("DEBUG: Found ID3v1 tag at offset %lu\n", tag->offset));
840
841 /* Convert ID3v1 data to ID3v2.4 */
842 tag->length = 128;
843 tag->version = 0;
844 tag->flags = 0;
845 tag->frames = NULL((void*)0);
846
847 pframe = &(tag->frames);
848
849 /* Add title field (offset 3, len 30). */
850 if (buf[3] != '\0') {
851 memcpy(sbuf, buf + 3, 30);
852 sbuf[30] = '\0';
853 *pframe = id3_make_frame("TIT2", "bs", 0, sbuf);
854 pframe = &((*pframe)->next);
855 }
856
857 /* Add artist field (offset 33, len 30). */
858 if (buf[33] != '\0') {
859 memcpy(sbuf, buf + 33, 30);
860 sbuf[30] = '\0';
861 *pframe = id3_make_frame("TPE1", "bs", 0, sbuf);
862 pframe = &((*pframe)->next);
863 }
864
865 /* Add album field (offset 63, len 30). */
866 if (buf[63] != '\0') {
867 memcpy(sbuf, buf + 63, 30);
868 sbuf[30] = '\0';
869 *pframe = id3_make_frame("TALB", "bs", 0, sbuf);
870 pframe = &((*pframe)->next);
871 }
872
873 /* Add release year (offset 93, len 4). */
874 if (buf[93] >= '0' && buf[93] <= '9' &&
875 buf[94] >= '0' && buf[94] <= '9' &&
876 buf[95] >= '0' && buf[95] <= '9' &&
877 buf[96] >= '0' && buf[96] <= '9') {
878 memcpy(sbuf, buf + 93, 4);
879 sbuf[4] = '\0';
880 *pframe = id3_make_frame("TDRC", "bs", 0, sbuf);
881 pframe = &((*pframe)->next);
882 }
883
884 /* Add comment (offset 97, len 30). */
885 if (buf[97] != '\0') {
886 memcpy(sbuf, buf + 97, 30);
887 sbuf[30] = '\0';
888 /* assume ISO-8859-1, unknown language, no description */
889 *pframe = id3_make_frame("COMM", "bssbs", 0, "XXX", "", 0, sbuf);
890 pframe = &((*pframe)->next);
891 }
892
893 /* Add track number (offset 126, len 1). */
894 if (buf[125] == '\0' && buf[126] != 0) {
895 sprintf(sbuf, "%u", buf[126]);
896 *pframe = id3_make_frame("TRCK", "bs", 0, sbuf);
897 pframe = &((*pframe)->next);
898 }
899
900 return 1;
901}

904/**
905 * Search for an ID3v2 tag at the beginning and end of the file.
906 * If no ID3v2 tag is found, search for an ID3v1 tag at the end
907 * of the file.
908 *
909 * Return 1 on success, 0 if no tag is found, or a negative error code
910 * if the tag can not be processed.
911 */
912static int id3_search_tag(FILE *f, struct ID3v2TagStruct *tag)
913{
914 unsigned char buf[32];
915 unsigned long pos, k, id3v1_pos = 0;
916 int j, ret;
917
918 /* Look for ID3v2 at the beginning of the file. */
919 fseek(f, 0, SEEK_SET0);
920 ret = id3_parse_v2_tag(f, tag);
921
922 if (ret == 0) {
923 /* Look for ID3v2 at the end of the file. */
924
925 fseek(f, 0, SEEK_END2);
926 pos = ftell(f);
927
928 while (pos > 128) {
929
930 /* Test for ID3v2 footer */
931 fseek(f, pos - 10, SEEK_SET0);
932 if (fread(buf, 1, 10, f) != 10)
933 return M3G_ERR_READ-6;
934 if (memcmp(buf, "3DI", 3) == 0 &&
935 buf[3] == 4 &&
936 ((buf[6] | buf[7] | buf[8] | buf[9]) & 0x80) == 0) {
937 /* Parse ID3v2.4 tag */
938 k = id3_get_syncsafe_int(buf + 6);
939 if (20 + k < pos) {
940 pos -= 20 + k;
941 fseek(f, pos, SEEK_SET0);
942 ret = id3_parse_v2_tag(f, tag);
943 break;
944 }
945 }
946
947 /* Test for ID3v1/Lyrics3v2 tag */
948 fseek(f, pos - 128, SEEK_SET0);
949 if (fread(buf, 1, 3, f) != 3)
950 return M3G_ERR_READ-6;
951 if (memcmp(buf, "TAG", 3) == 0) {
952 /* Skip over ID3v1 tag and continue */
953 pos -= 128;
954 DBG(("DEBUG: Skipping ID3v1 tag at offset %lu\n", pos));
955 id3v1_pos = pos;
956 /* Test for Lyrics3v2 tag */
957 if (pos > 26) {
958 fseek(f, pos - 15, SEEK_SET0);
959 if (fread(buf, 1, 15, f) != 15)
960 return M3G_ERR_READ-6;
961 if (memcmp(buf + 6, "LYRICS200", 9) == 0) {
962 /* Skip over Lyrics tag */
963 k = 0;
964 for (j = 0; j < 6; j++) {
965 if (buf[j] < '0' || buf[j] > '9') {
966 k = 0;
967 break;
968 }
969 k = 10 * k + (buf[j] - '0');
970 }
971 if (k >= 11 && k + 15 < pos) {
972 pos -= k + 15;
973 DBG(("DEBUG: Skipping Lyrics3v2 tag at offset %lu\n", pos));
974 }
975 }
976 }
977 continue;
978 }
979
980 /* Test for APE tag */
981 fseek(f, pos - 32, SEEK_SET0);
982 if (fread(buf, 1, 32, f) != 32)
983 return M3G_ERR_READ-6;
984 if (memcmp(buf, "APETAGEX", 8) == 0) {
985 /* Skip over APE tag and continue */
986 k = buf[12] | (buf[13] << 8) | (buf[14] << 16) | (buf[15] << 24); /* tag length */
987 if (buf[23] & 0x80)
988 k += 32; /* tag header present */
989 if (k >= 32 && k < pos) {
990 pos -= k;
991 DBG(("DEBUG: Skipping APE tag at offset %lu\n", pos));
992 continue;
993 }
994 }
995
996 /* No more tags to skip. */
997 break;
998
999 }
1000
1001 }
1002
1003 if (ret == 0 && id3v1_pos != 0) {
1004 /* Read ID3v1 at the end of the file. */
1005 fseek(f, id3v1_pos, SEEK_SET0);
1006 ret = id3_parse_v1_tag(f, tag);
1007 }
1008
1009 return ret;
1010}

1013/**
1014 * Seek to offset in inf and copy count bytes to the current
1015 * position in outf. If count == -1, copy until the end of the file.
1016 *
1017 * Return 1 on success, otherwise a negative error code.
1018 */
1019static int id3_copy_data(FILE *inf, FILE *outf, long offset, long count)
1020{
1021 const long bufsize = 65536;
1022 char *buf;
1023 size_t k;
1024 int ret;
1025
1026 ret = 1;
1027 buf = malloc(bufsize);
1028
1029 if (fseek(inf, offset, SEEK_SET0))
1030 ret = M3G_ERR_READ-6;
1031
1032 while (ret == 1 && count != 0) {
1033 k = (count > 0 && count < bufsize) ? count : bufsize;
1034 k = fread(buf, 1, k, inf);
1035 if (k == 0) {
1036 if (!feof(inf) || count > 0)
1037 ret = M3G_ERR_READ-6;
1038 break;
1039 }
1040 if (fwrite(buf, 1, k, outf) != k) {
1041 ret = M3G_ERR_WRITE-7;
1042 break;
1043 }
1044 if (count > 0)
1045 count -= k;
1046 }
1047
1048 free(buf);
1049 return ret;
1050}

1091/**
1092 * (Re-)Write gain information to an ID3v2 tag.
1093 *
1094 * Writes or updates an ID3v2.4 tag at the beginning of the file.
1095 * If the file does not yet contain an ID3v2 tag, a new tag is created,
1096 * copying basic fields from an ID3v1 tag if present.
1097 * If the file contains an ID3v2.2 or ID3v2.3 tag, it is rewritten as ID3v2.4.
1098 *
1099 * Since modifications are made at the beginning of the file, the entire
1100 * file is rewritten to a temporary file and then moved in place of the
1101 * old file.
1102 */
1103int WriteMP3GainID3Tag(char *filename, struct MP3GainTagInfo *info, int saveTimeStamp)
1104{
1105 char sbuf[32];
1106 char *tmpfilename;
1107 FILE *f, *outf;
1108 struct ID3v2TagStruct tag;
1109 struct ID3v2FrameStruct *frame, **pframe;
1110 int ret, need_update;
1111
1112 if (saveTimeStamp)
1113 fileTime(filename, storeTime);
1114
1115 f = fopen(filename, "rb");
1116 if (f == NULL((void*)0)) {
1117 passError(MP3GAIN_UNSPECIFED_ERROR01, 3, "Could not open ", filename, "\n");
1118 return M3G_ERR_FILEOPEN-5;
1119 }
1120 ret = id3_search_tag(f, &tag);
1121
1122 switch (ret) {
1123 case M3G_ERR_READ-6:
1124 passError(MP3GAIN_UNSPECIFED_ERROR01, 3, "Error reading ", filename, "\n");
1125 break;
1126 case M3G_ERR_TAGFORMAT-8:
1127 passError(MP3GAIN_UNSPECIFED_ERROR01, 3, "Unsupported ID3v2 tag in ", filename, "\n");
1128 break;
1129 }
1130
1131 if (ret < 0) {
1132 /* Error. */
1133 fclose(f);
1134 return ret;
1135 }
1136
1137 if (ret == 0) {
1138 /* No ID3v2 or ID3v1 tag found in the file; create a new tag. */
1139 tag.offset = 0;
1140 tag.length = 0;
1141 tag.version = 0;
1142 tag.flags = 0;
1143 tag.frames = NULL((void*)0);
1144 }
1145
1146 /* Kill existing replaygain frames. */
1147 need_update = 0;
1148 pframe = &(tag.frames);
1149 while ((frame = *pframe)) {
1150 if (id3_decode_rva2_frame(frame, NULL((void*)0)) == 1 ||
1151 id3_decode_mp3gain_frame(frame, NULL((void*)0)) == 1) {
1152 /* This is a ReplayGain frame; kill it. */
1153 need_update = 1;
1154 *pframe = frame->next;
1155 free(frame->data);
1156 free(frame);
1157 } else {
1158 pframe = &((*pframe)->next);
1159 }
1160 }
1161
1162 /* Append new replaygain frames. */
1163 if (info->haveTrackGain) {
1164 need_update = 1;
1165 frame = id3_make_rva2_frame(0, info->trackGain, info->haveTrackPeak, info->trackPeak);
1166 *pframe = frame;
1167 pframe = &(frame->next);
1168 }
1169 if (info->haveAlbumGain) {
1170 need_update = 1;
1171 frame = id3_make_rva2_frame(1, info->albumGain, info->haveAlbumPeak, info->albumPeak);
1172 *pframe = frame;
1173 pframe = &(frame->next);
1174 }
1175
1176 /* Append mp3gain-specific frames. */
1177 if (info->haveMinMaxGain) {
1178 need_update = 1;
1179 sprintf(sbuf, "%03d,%03d", info->minGain, info->maxGain);
1180 frame = id3_make_frame("TXXX", "bsbs", 0, "MP3GAIN_MINMAX", 0, sbuf);
1181 *pframe = frame;
1182 pframe = &(frame->next);
1183 }
1184 if (info->haveAlbumMinMaxGain) {
1185 need_update = 1;
1186 sprintf(sbuf, "%03d,%03d", info->albumMinGain, info->albumMaxGain);
1187 frame = id3_make_frame("TXXX", "bsbs", 0, "MP3GAIN_ALBUM_MINMAX", 0, sbuf);
1188 *pframe = frame;
1189 pframe = &(frame->next);
1190 }
1191 if (info->haveUndo) {
1192 need_update = 1;
1193 sprintf(sbuf, "%+04d,%+04d,%c", info->undoLeft, info->undoRight, info->undoWrap ? 'W' : 'N');
1194 frame = id3_make_frame("TXXX", "bsbs", 0, "MP3GAIN_UNDO", 0, sbuf);
1195 *pframe = frame;
1196 pframe = &(frame->next);
1197 }
1198
1199 if (!need_update) {
1200 /* No need to change MP3 file. */
1201 fclose(f);
1202 id3_release_frames(tag.frames);
1203 return 0;
1204 }
1205
1206 /* Create temporary file. */
1207 tmpfilename = malloc(strlen(filename) + 5);
1208 strcpy(tmpfilename, filename);
1209 strcat(tmpfilename, ".TMP");
1210 outf = fopen(tmpfilename, "wb");
1211 if (outf == NULL((void*)0)) {
1212 passError(MP3GAIN_UNSPECIFED_ERROR01, 3, "Can not create temporary file ", tmpfilename, "\n");
1213 fclose(f);
1214 free(tmpfilename);
1215 id3_release_frames(tag.frames);
1216 return M3G_ERR_CANT_MAKE_TMP-2;
1217 }
1218
1219 /* Write new ID3v2 tag. */
1220 ret = id3_write_tag(outf, &tag);
1221
1222 /* Write rest of MP3 file. */
1223 if (ret >= 0) {
1224 if (tag.version == 0) {
1225 /* The original file has no ID3v2 tag; copy everything. */
1226 ret = id3_copy_data(f, outf, 0, -1);
1227 } else {
1228 /* The original file has an ID3v2 tag; copy everything
1229 except the original tag. */
1230 ret = id3_copy_data(f, outf, 0, tag.offset);
1231 if (ret >= 0)
1232 ret = id3_copy_data(f, outf, tag.offset + tag.length, -1);
1233 }
1234 }
1235
1236 fclose(outf);
1237 fclose(f);
1238 id3_release_frames(tag.frames);
1239
1240 switch (ret) {
1241 case M3G_ERR_READ-6:
1242 passError(MP3GAIN_UNSPECIFED_ERROR01, 3, "Error reading ", filename, "\n");
1243 break;
1244 case M3G_ERR_WRITE-7:
1245 passError(MP3GAIN_UNSPECIFED_ERROR01, 3, "Error writing ", tmpfilename, "\n");
1246 break;
1247 }
1248
1249 if (ret < 0) {
1250 /* Delete temp file after error. */
1251 remove(tmpfilename);
1252 free(tmpfilename);
1253 return ret;
1254 }
1255
1256 /* Replace original file. */
1257#ifdef WIN32
1258 /* "rename" function in WIN32 does _not_ replace existing destination file, so
1259 we have to manually remove it first */
1260 if (remove(filename) != 0) {
1261 passError(MP3GAIN_UNSPECIFED_ERROR01, 5, "Can not rename ", tmpfilename, " to ", filename, "\n");
1262 ret = M3G_ERR_RENAME_TMP-4;
1263 /* Do NOT remove the temp file itself, just in case the "remove(filename)" function
1264 only _temporarily_ failed, and the original file will disappear soon, such as when
1265 all handles on the file are closed. If it does disappear and we also
1266 delete the tmp file, then the file is completely gone... */
1267 free(tmpfilename);
1268 }
1269#endif
1270
1271 ret = 1;
1272 if (rename(tmpfilename, filename)) {
1273 remove(tmpfilename);
1274 passError(MP3GAIN_UNSPECIFED_ERROR01, 5, "Can not rename ", tmpfilename, " to ", filename, "\n");
1275 ret = M3G_ERR_RENAME_TMP-4;
1276 } else {
1277 if (saveTimeStamp)
1278 fileTime(filename, setStoredTime);
1279 }
1280
1281 free(tmpfilename);
1282
1283 return ret;
1284}
